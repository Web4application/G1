🤖 6. REAL MOTOR CONTROL NODE
```py
robot_control/motor_node.py

import rclpy
from rclpy.node import Node
from std_msgs.msg import String

class MotorNode(Node):
    def __init__(self):
        super().__init__('motor_node')

        self.sub = self.create_subscription(
            String,
            'cmd_vel',
            self.execute,
            10
        )

    def execute(self, msg):
        command = msg.data

        if command == "forward":
            self.get_logger().info("Motor: forward")
        elif command == "left":
            self.get_logger().info("Motor: left")
        elif command == "right":
            self.get_logger().info("Motor: right")
        else:
            self.get_logger().info("Motor: stop")

```
