🤖 4. FINAL AI BRAIN (DQN + GPT FUSION)

class CognitiveBrain:
    def __init__(self, dqn, gpt):
        self.dqn = dqn
        self.gpt = gpt

    def decide(self, state, uncertainty):
        action = self.dqn.act(state)

        if uncertainty > 0.7:
            action = self.gpt(state, action)

        return action
